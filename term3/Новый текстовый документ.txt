class MinNode
{
public:
    char data;

    unsigned freq;

    MinNode* left, * right;

    MinNode(char data, unsigned freq)
    {
        left = right = nullptr;
        this->data = data;
        this->freq = freq;
    }
};

struct compare
{
    bool operator()(MinNode* l, MinNode* r)
    {
        return (l->freq > r->freq);
    }
};
